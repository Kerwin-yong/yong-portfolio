<script>
function changeLang(lang) {
    const elements = document.querySelectorAll('.lang');
    elements.forEach(el => {
        el.innerText = el.getAttribute('data-' + lang);
    });
    localStorage.setItem('prefLang', lang);
}
window.onload = () => { changeLang(localStorage.getItem('prefLang') || 'zh'); };
